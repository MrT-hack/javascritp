//Función para sumar dos números
function suma(a, b) {
  return a + b;
}

//Función para restar dos números
function resta(a, b) {
  return a - b;
}

//Función para multiplicar dos números
function multiplicacion(a, b) {
  return a * b;
}

//Función para dividir dos números
function division(a, b) {
  return a / b;
}
